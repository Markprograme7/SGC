function ligarDesliga(){ var imagem = document.getElementById('lamp').src; var imagem_ligado = 'images/lampada-on.jpg'; var imagem_desligado = 'images/lampada-off.jpg'; if(imagem == imagem_ligado){ document.getElementById('lamp').src = imagem_desligado; }else{ document.getElementById('lamp').src = imagem_ligado; } } document.getElementById("lamp").addEventListener("click", ligarDesliga);
